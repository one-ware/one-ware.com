export function PartnerLogo({ name, href, img }) {
  const mask = `url(${img}) center / contain no-repeat`;

  return (
    <a
      href={href}
      target="_blank"
      rel="noopener noreferrer"
      aria-label={`Visit our partner ${name}`}
      style={{ display: "inline-block" }}
    >
      <div
        role="img"
        aria-label={`Logo ${name}`}
        style={{
          width: 300,
          height: 150,
          backgroundColor: "#ffffffff",
          WebkitMask: mask,
          mask: mask,
          cursor: "pointer",
          flexShrink: 0,
        }}
      />
    </a>
  );
}

export function PartnerSection({ name, description, href, img, logoLeft = false }) {
  return (
    <div style={{ display: 'flex', alignItems: 'center', gap: '3rem', flexWrap: 'wrap', flexDirection: logoLeft ? "row-reverse" : "row", marginBottom: '4rem' }}>
      <div style={{ flex: 1, whiteSpace: "pre-line" }}>
          <h2>{name}</h2>
          {description}
      </div>
      <PartnerLogo
        href={href}
        img={img}
        name={name}
        />
    </div>
  );
}
